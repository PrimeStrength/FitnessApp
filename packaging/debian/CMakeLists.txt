set(CPACK_DEBIAN_PACKAGE_NAME ${PROJECT_NAME})
set(CPACK_DEBIAN_PACKAGE_SHLIBDEPS ON)

add_custom_command(
  OUTPUT "${CMAKE_BINARY_DIR}/${PROJECT_NAME}-${PROJECT_VERSION}-Linux.deb"
  COMMAND cpack -G DEB
  WORKING_DIRECTORY ${CMAKE_BINARY_DIR}
  DEPENDS web ${PROJECT_NAME}
)

add_custom_target(deb
  DEPENDS "${CMAKE_BINARY_DIR}/${PROJECT_NAME}-${PROJECT_VERSION}-Linux.deb"
)
